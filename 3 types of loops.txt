1.for loop:

eg1:// Print numbers from 1 to 10
#include <stdio.h>

int main() {
  int i;

  for (i = 1; i < 11; ++i)
  {
    printf("%d ", i);
  }
  return 0;
}

Also, we can skip the initial value expression, 
condition and/or increment by adding a semicolon.

For example:
#include<stdio.h>
int main()
{
int i=0;
int max = 10;
for (; i < max; i++)
{
  		printf("%d\n", i);
	}
	return 0;
}

Eg3 of for loop:


// Program to calculate the sum of first n natural numbers

#include <stdio.h>
int main()
{
    int num, count, sum = 0;

    printf("Enter a positive integer: ");
    scanf("%d", &num);

    // for loop terminates when num is less than count
    for(count = 1; count <= num; count++)
    {
        sum += count;
    }

    printf("Sum = %d", sum);

    return 0;
}

2.While loop:
eg1:
#include <stdio.h>

int main()
{
	// initialization expression
	int i = 1;

	// test expression
	while (i < 6)
	{
		printf( “Thundersoft\n");	

		// update expression
		i++;
	}

	return 0;
}

3.do-while loop:
eg1:

#include <stdio.h>

int main()
{
	int i = 2; // Initialization expression

	do
	{
		// loop body
		printf( "Thundersoft\n");	

		// update expression
		i++;

	} while (i < 1); // test expression

	return 0;
}
